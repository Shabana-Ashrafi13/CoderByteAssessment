def get_type_of_business(input_n):
	

	if input_n <= 0:

		return None

	elif input_n < 25:

		return "STANDARD"

	elif input_n >= 25:

		return "CORPORATE"

	else:

		return "INVALID"



if __name__ == "__main__":


	input_num = int(input("Enter a number to get the type of the business"))

	print(get_type_of_business(input_num))

Class I : value<0 => Invalid Class
